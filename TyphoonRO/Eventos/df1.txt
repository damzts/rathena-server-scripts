-	script	Killa_warp	-1,{

OnPCDieEvent:
	getmapxy(.@map$,.@x,.@y,0);
	if(.@map$=="pvp_n_1-3")
	{
		set #kill_point, 0;
		announce "["+strcharinfo(0)+"] murio, queda fuera del evento!.",bc_all;
		warp "prontera",0,0;
		end;
	}
OnPCKillEvent:
	getmapxy(.@map$,.@x,.@y,0);
	if(.@map$=="pvp_n_1-3")
	{
	set #kill_point,#kill_point+1;
	if ( #kill_point % 4 == 0) {
	announce "["+strcharinfo(0)+"] ha matado 4 jugadores consecutivos! Obtendra un premio especial",bc_all;
	getitem 607, 20;
	end;
	}
	}
}
